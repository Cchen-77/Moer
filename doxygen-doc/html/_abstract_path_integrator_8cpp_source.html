34Spectrum AbstractPathIntegrator::Li(const Ray &initialRay, std::shared_ptr<Scene> scene)
35{
36 Spectrum L(0.0);
37 Spectrum throughput(1.0);
38 Ray ray = initialRay;
39 double pdfLastScatterSample = 1e99;
40 bool isLastScatterSampleDelta = false;
41 int nBounce = 0;
42
43 while (true)
44 {
45 std::optional<Intersection> itsOpt = scene->intersect(ray);
46
47 // EVAL EMITTANCE
48 PathIntegratorLocalRecord evalLightRecord = evalEmittance(scene, itsOpt, ray);
49
50 if (evalLightRecord.f.isBlack() == false)
51 {
52 double misw = MISWeight(pdfLastScatterSample, evalLightRecord.pdf);
53 if (isLastScatterSampleDelta)
54 {
55 misw = 1.0;
56 }
57 L += throughput * evalLightRecord.f * misw;
58 }
59
60 // TERMINATE IF NO INTERSECTION
61 if (!itsOpt.has_value())
62 {
63 break;
64 }
65
66 Intersection its = itsOpt.value();
67
68 // RUSSIAN ROULETTE
69 nBounce++;
70 double pSurvive = russianRoulette(scene, its, throughput, nBounce);
71 if (randFloat() > pSurvive)
72 break;
73 throughput /= pSurvive;
74
75 // SAMPLE DIRECT LIGHTING
76 // Support multiple direct light samples per intersection. n=1 by default.
77 for (int i = 0; i < nDirectLightSamples; i++)
78 {
79 PathIntegratorLocalRecord sampleLightRecord = sampleDirectLighting(scene, its, ray);
80 PathIntegratorLocalRecord evalScatterRecord = evalScatter(scene, its, ray, sampleLightRecord.wi);
81
82 if (sampleLightRecord.f.isBlack() == false)
83 {
84 double misw = MISWeight(sampleLightRecord.pdf, evalScatterRecord.pdf);
85 if (sampleLightRecord.isDelta)
86 {
87 misw = 1.0;
88 }
89 L += throughput * sampleLightRecord.f * evalScatterRecord.f / sampleLightRecord.pdf * misw / nDirectLightSamples;
90 }
91 }
92
93 // SAMPLE SCATTER (BSDF or PHASE)
94 PathIntegratorLocalRecord sampleScatterRecord = sampleScatter(scene, its, ray);
95 pdfLastScatterSample = sampleScatterRecord.pdf;
96 isLastScatterSampleDelta = sampleScatterRecord.isDelta;
97 if (sampleScatterRecord.f.isBlack() == false)
98 {
99 throughput *= sampleScatterRecord.f / sampleScatterRecord.pdf;
100 }
101 else
102 {
103 break;
104 }
105 const double eps = 1e-4; // todo
106 ray = Ray(its.position + sampleScatterRecord.wi * eps, sampleScatterRecord.wi);
107 }
108
109 return L;
110}
