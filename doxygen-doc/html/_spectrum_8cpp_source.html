29// @brief calculate the average value from samples within interval [lambdaBegin,lambdaEnd].
30// @param samples (maybe unsorted) spectrum samples.
31// @param lambdaBegin begin lambda.
32// @param lambdaEnd end lambda.
33// @return average value of samples within intraval [lambdaBegin,lambdaEnd].
34double averageSpectrumSamples(std::vector<SpectrumSample> samples, double lambdaBegin, double lambdaEnd)
35{
36 auto len = samples.size();
37 // handle boundary situation
38 if (lambdaEnd <= samples[0].lambda) return samples[0].value;
39 if (lambdaBegin >= samples[len - 1].lambda) return samples[len - 1].value;
40 if (len == 1) return samples[0].value;
41
42 // handle begin & end
43 double sum = 0;
44 if (lambdaBegin < samples[0].lambda) sum += samples[0].value * (samples[0].lambda - lambdaBegin);
45 if (samples[len - 1].lambda < lambdaEnd) sum += samples[len - 1].value * (lambdaEnd - samples[len - 1].lambda);
46
47 auto interp = [samples](double w, int i) {
48 // Linear interpolation
49 return mathLerp((w - samples[i].lambda) / (samples[i + 1].lambda - samples[i].lambda), samples[i].value, samples[i + 1].value);
50 };
51 int i = 0;
52 while (lambdaBegin > samples[i + 1].lambda) i++;
53 for (; i + 1 < len && lambdaEnd >= samples[i].lambda; i++) {
54 double segLambdaBegin = std::max(lambdaBegin, samples[i].lambda);
55 double segLambdaEnd = std::min(lambdaEnd, samples[i + 1].lambda);
56 // area of a trapezium
57 sum += 0.5 * (interp(segLambdaBegin, i) + interp(segLambdaEnd, i)) *
58 (segLambdaEnd - segLambdaBegin);
59 }
60 return sum / (lambdaEnd - lambdaBegin);
61}
