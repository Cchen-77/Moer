28{
29 auto R = radius;

31 auto o = r.origin;
32 auto d = r.direction;
33 auto u = o - C;
34 double a = d.length2();

36 double c = u.length2() - R * R;
37 double delta = b * b - 4 * a * c;
38 if (delta < 0)
39 return std::nullopt;
40 double sqrtDelta = sqrt(delta);
41 double t1 = (-b + sqrtDelta) / (2.0 * a);
42 double t2 = (-b - sqrtDelta) / (2.0 * a);
43 bool flag = false;
44 Intersection ans;
45 double t = INFINITY;
46 if (t1 >= 0 && t1 < t && t1 >= r.timeMin && t1 <= r.timeMax)
47 {
48 t = t1;
49 Vec3d n = o + d * t1 - C;
50 n = normalize(n);
51 ans.position = o + d * t1;
52 ans.geometryNormal = n;
53 ans.geometryTangent = normalize(Vec3d(n.z, 0, -n.x));
