34void MonteCarloIntegrator::renderPerThread(std::shared_ptr<Scene> scene)
35{
36 while(true)
37 {
38 auto optionalTile=tileGenerator->generateNextTile();
39 if(optionalTile==std::nullopt)
40 break;
41 auto tile=optionalTile.value();
42
43 for(auto it=tile->begin();it!=tile->end();++it)
44 {
45 auto pixelPosition=*it;
46
47 const auto &cam = *this->camera;
48 sampler->startPixel(pixelPosition);
49 for (int i = 0; i < spp; i++)
50 {
51 auto L = Li(
52 cam.generateRay(
53 film->getResolution(),
54 pixelPosition,
55 sampler->getCameraSample()
56 ), scene
57 );
58 film->deposit(pixelPosition, L);
59 sampler->nextSample();
60 }
61 }
62 }
63}
