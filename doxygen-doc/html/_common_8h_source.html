55inline float fresnel(float cosThetaI, float extIOR, float intIOR) {
56 float etaI = extIOR, etaT = intIOR;
57
58 if (extIOR == intIOR)
59 return 0.0f;
60
61 /* Swap the indices of refraction if the interaction starts
62 at the inside of the object */
63 if (cosThetaI < 0.0f) {
64 std::swap(etaI, etaT);
65 cosThetaI = -cosThetaI;
66 }
67
68 /* Using Snell's law, calculate the squared sine of the
69 angle between the normal and the transmitted ray */
70 float eta = etaI / etaT,
71 sinThetaTSqr = eta*eta * (1-cosThetaI*cosThetaI);
72
73 if (sinThetaTSqr > 1.0f)
74 return 1.0f; /* Total internal reflection! */
75
76 float cosThetaT = std::sqrt(1.0f - sinThetaTSqr);
77
78 float Rs = (etaI * cosThetaI - etaT * cosThetaT)
79 / (etaI * cosThetaI + etaT * cosThetaT);
80 float Rp = (etaT * cosThetaI - etaI * cosThetaT)
81 / (etaT * cosThetaI + etaI * cosThetaT);
82
83 return (Rs * Rs + Rp * Rp) / 2.0f;
84}
