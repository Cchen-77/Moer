54template<typename T>
55void shuffle(std::vector<T> &samples, RandomNumberGenerator &rng) {
56 int count = samples.size();
57 for (int i = 0; i < count; ++i) {
58 int other = rng(i, count);
59 std::swap(samples[i], samples[other]);
60 }
61}
