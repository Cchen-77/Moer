138std::pair<std::shared_ptr<Light>, double> PathIntegrator::chooseOneLight(std::shared_ptr<Scene> scene,
139 const Intersection &its,
140 const Ray &ray,
141 double lightSample)
142{
143 // uniformly weighted
144 std::shared_ptr<std::vector<std::shared_ptr<Light>>> lights = scene->getLights();
145 int numLights = lights->size();
146 int lightID = std::min(numLights - 1, (int)(lightSample * numLights));
147 std::shared_ptr<Light> light = lights->operator[](lightID);
148 return {light, 1.0 / numLights};
