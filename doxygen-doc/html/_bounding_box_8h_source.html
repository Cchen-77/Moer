52 std::optional<Point2d> Intersection(const Ray& r) {
53 double t0 = 0, t1 = DBL_MAX;
54 for (int i = 0; i < 3; i++) {
55 double inv = 1.0 / r.direction[i];
56 double tNear = (pMin[i] - r.origin[i]) * inv;
57 double tFar = (pMax[i] - r.origin[i]) * inv;
58 if (tNear > tFar) std::swap(tNear, tFar);
59 if (t0 < tNear) t0 = tNear;
60 if (t1 > tFar) t1 = tFar;
61 }
62 if (t0 > t1) return std::nullopt;
63 return std::make_optional(Point2d(t0, t1));
64 }

69 double SurfaceArea() {
70 double x = pMax[0] - pMin[0];
71 double y = pMax[1] - pMin[1];
72 double z = pMax[2] - pMin[2];
73 return 2 * (x * y + x * z + y * z);
74 }
